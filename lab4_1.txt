#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>

int main() {
    pid_t pid;

    pid = fork();

    if (pid == -1) {
        perror("fork failed");
        exit(1);
    }
    else if (pid == 0) { // child process
        execl("/bin/ls", "ls", "-la", NULL);
    }
    else { // parent process
        wait(NULL);
        printf("Child process complete.\n");
    }

    return 0;
}
